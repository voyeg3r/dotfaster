# Introdução

 Loop for, used to perform repetitive tasks

# creating many folders

    for ((i=1;i<=7;i++)){
        mkdir $(printf "%03d\n" $i)
    }

    for i in $(echo {001..010}); do
        mkdir folder-$i
    done

We can also create folders this way

     mkdir -p folder-{001..007}
