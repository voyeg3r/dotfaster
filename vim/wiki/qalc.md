    qalc 62499836 byte to gibibyte
    62499836 * byte = approx. 0.058207508 gibibyte

    qalc 40 degree celsius to fahrenheit
    (40 * degree) * celsius = 104 deg*oF

    qalc 22 degree celsius to fahrenheit | awk '{print $7}'
